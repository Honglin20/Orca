## D2. MCP server 骨架（FastMCP + stdio + flush）

### D2.1 `orca/iface/mcp/transport.py`（新文件）

```python
"""transport.py —— stdio writer flush 兜底（SPEC §4.1）。

规避 opencode flush bug (#21516)：每条 JSON-RPC 消息写完显式 flush。
即使 SDK 未来某版本忘了 flush，Orca 不受影响。
"""
class FlushingStdoutWriter:
    """包装 stdout，每写一次 flush。"""
    def __init__(self, stream): self._stream = stream
    async def write(self, data: bytes) -> None:
        self._stream.write(data)
        self._stream.flush()
```

实现要点：
- 不替换 SDK 的 stdio_server，只在 write 层包一层
- 兼容 bytes / str 输入
- 单测用 io.StringIO / BytesIO mock 验证 flush 调用次数

不含编排逻辑（manager 才是托管入口）。
"""
from mcp.server.fastmcp import FastMCP

class OrcaMcpServer:

    def __init__(self, manager: RunManager):

        """阻塞跑 stdio，stdin EOF 退出（SPEC §1.3）。"""

def run_mcp_server(*, with_web: bool, web_port: int, max_concurrent: int, idle_timeout: int):
    """入口（commands.py 调）。SPEC §1.4 单例 assert。"""
    manager = RunManager(max_concurrent=max_concurrent)
    _assert_singleton(manager)  # 进程内 RunManager 实例数 == 1

def start_workflow(yaml_path: str, inputs: dict | None = None,

@mcp.tool()
def get_task_status(task_id: str) -> dict:

    summary = manager.run_summary(task_id)

### D3.3 `resolve_gate`

```python
@mcp.tool()
def resolve_gate(task_id: str, gate_id: str, decision: str) -> dict:
    """SPEC §2.2。复用 routes/gate.py 同款多 run 分发逻辑。"""
    handle = manager.get_handle(task_id)
    if handle is None:
        return {"ok": False, "_hint": "Unknown task_id."}
    ok = handle.gate_handler.resolve(gate_id, decision, source="mcp")
    return {
        "ok": ok,
        "status": "running" if ok else "needs_decision",
        "_hint": hints.after_resolve(ok),
    }
```

要点：
- **多 run 分发逻辑直接用 run_id**（MCP 工具入参带 task_id，不需要 session_id 反查——比 routes/gate.py 更简单）
- `source="mcp"` 写死（MCP 壳标识）
- 复用 `handler.resolve`（phase-6 + phase-9a），零新代码路径

### D4.3 stdin EOF 生命周期（SPEC §1.3）

`orca/iface/mcp/server.py` `run_mcp_server` 实现两个分支：

- 无 `--with-web`：`run_stdio` 返回（stdin EOF）→ 进程 drain in-flight tool call（最大 5s）→ cancel 后台 run task → 退出
- 有 `--with-web`：`run_stdio` 返回 → 启动 `_wait_for_idle_or_signal` → 无活跃 run 持续 `idle_timeout` 分钟 OR SIGINT → 退出

@pytest.mark.asyncio
async def test_e2e_gate_workflow(orca_mcp_inprocess):
    """SPEC §6.3 E2E-2：合成 gate，验证 resolve_gate 全流程。"""
    manager, server = orca_mcp_inprocess

    # 1. 起 demo_linear run（后台跑，会很快 completed —— 我们在中途注入 gate）
    run_id = await manager.start_run("examples/demo_linear.yaml")
    handle = manager.get_handle(run_id)

    # 2. 测试 helper：后台起一个 handler.request（模拟 node 触发 gate）
    async def fire_gate():
        gate = HumanGate(id="test_gate", prompt="批准？", context={},
                         source="agent_ask", run_id=run_id, node="b",
                         options=["yes", "no"])
        return await handle.gate_handler.request(gate)
    gate_task = asyncio.create_task(fire_gate())

    # 3. 等tape写入 requested 事件
    await asyncio.sleep(0.2)

    # 4. MCP get_task_status 应见 needs_decision
    status = await server.get_task_status(task_id=run_id)
    assert status["status"] == "needs_decision"
    assert status["gate"]["gate_id"] == "test_gate"

    result = await server.resolve_gate(task_id=run_id, gate_id="test_gate", decision="yes")
    assert result["ok"] is True

    # 6. gate_task 完成，answer == "yes"
    answer, source = await gate_task
    assert answer == "yes"
    assert source == "mcp"  # SPEC §0.1 第五条

    # 7. 后续 status 应不再 needs_decision
    status = await server.get_task_status(task_id=run_id)
    assert status["status"] != "needs_decision"

### D5.4 E2E-3：跨壳 race（`test_e2e_cross_shell.py`，CI 可跑）

```python
@pytest.mark.asyncio
async def test_e2e_cross_shell_race(orca_mcp_inprocess):
    """SPEC §6.3 E2E-3：MCP + Web 同时连，Web 答门，MCP poll 跳过 needs_decision。"""
    manager, server = orca_mcp_inprocess
    run_id = await manager.start_run("examples/demo_linear.yaml")
    handle = manager.get_handle(run_id)

    # 1. fire gate
    gate_task = asyncio.create_task(_fire_gate(handle, run_id))
    await asyncio.sleep(0.2)

    # 2. Web 答（直接调 handler.resolve，模拟 POST /gate/respond）
    web_ok = handle.gate_handler.resolve("test_gate", "no", source="web")
    assert web_ok is True

    # 3. MCP 也答（晚到）—— 应输
    mcp_ok = handle.gate_handler.resolve("test_gate", "yes", source="mcp")
    assert mcp_ok is False

    # 4. gate_task 收到的 source 应是 web（赢家）
    answer, source = await gate_task
    assert source == "web"

    # 5. MCP get_task_status 不再见 needs_decision（resolved 写 tape）
    status = await server.get_task_status(task_id=run_id)
    assert status["status"] != "needs_decision"
```
